    \chapter{Lagrange Interpolation}
    Given $n+1$ points $(x_0, y_0)$, ... $(x_n, y_n)$, the Lagrange Interpolation is given by 
    \begin{align*}
        f_n(x) = \sum^{n}_{i=0} L_i(x)f(x_i)
    \end{align*}
    where $f(x_i) = y_i$ and $L_i(x)=\prod_{j=0,j\ne i}^{n}\frac{x-x_j}{x_i-x_j}$.
\end{appendices}
